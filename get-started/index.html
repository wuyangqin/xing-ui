    # 添加 CSS 样式

    使用本框架前，请确保整个项目的 CSS 开启了 border-box

    *,*::before,*::after { box-sizing: border-box; }

    # 按需引入

    import { XButton } from 'xing-ui'
import 'xing-ui/dist/xing.css'
import Vue from 'vue'
                 
new Vue({
  el: '#app',
  components: {
    XButton
  }
})
